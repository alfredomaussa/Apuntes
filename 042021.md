import React, { Component } from 'react';

class App extends Component {
  constructor() {
    super();
    this.state = {color: "red"};
  }
  changeColor = () => {
    this.setState({color: "blue"});
  }
  render() {
    return (
      <div className="App">
        <h1>Hello World! {this.state.color}</h1>
        <button onClick={()=> this.setState({color:"green"})}>press</button>
      </div>
    );
  }
}

export default App;
